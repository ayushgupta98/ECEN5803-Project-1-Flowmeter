680/* Interrupt Priorities are WORD accessible only under ARMv6M */
681/* The following MACROS handle generation of the register offset and byte masks */
682#define _BIT_SHIFT(IRQn) ( ((((uint32_t)(int32_t)(IRQn)) ) & 0x03UL) * 8UL)
683#define _SHP_IDX(IRQn) ( (((((uint32_t)(int32_t)(IRQn)) & 0x0FUL)-8UL) >> 2UL) )
684#define _IP_IDX(IRQn) ( (((uint32_t)(int32_t)(IRQn)) >> 2UL) )

693__STATIC_INLINE void NVIC_EnableIRQ(IRQn_Type IRQn)
694{
695 NVIC->ISER[0] = (uint32_t)(1UL << (((uint32_t)(int32_t)IRQn) & 0x1FUL));
696}

698
705__STATIC_INLINE void NVIC_DisableIRQ(IRQn_Type IRQn)
706{
707 NVIC->ICER[0] = (uint32_t)(1UL << (((uint32_t)(int32_t)IRQn) & 0x1FUL));
708 __DSB();
709 __ISB();
710}

724{
725 return((uint32_t)(((NVIC->ISPR[0] & (1UL << (((uint32_t)(int32_t)IRQn) & 0x1FUL))) != 0UL) ? 1UL : 0UL));

762__STATIC_INLINE void NVIC_SetPriority(IRQn_Type IRQn, uint32_t priority)
763{
764 if((int32_t)(IRQn) < 0) {
765 SCB->SHP[_SHP_IDX(IRQn)] = ((uint32_t)(SCB->SHP[_SHP_IDX(IRQn)] & ~(0xFFUL << _BIT_SHIFT(IRQn))) |
766 (((priority << (8 - __NVIC_PRIO_BITS)) & (uint32_t)0xFFUL) << _BIT_SHIFT(IRQn)));
767 }
768 else {
769 NVIC->IP[_IP_IDX(IRQn)] = ((uint32_t)(NVIC->IP[_IP_IDX(IRQn)] & ~(0xFFUL << _BIT_SHIFT(IRQn))) |
770 (((priority << (8 - __NVIC_PRIO_BITS)) & (uint32_t)0xFFUL) << _BIT_SHIFT(IRQn)));
771 }
772}
773
774
786__STATIC_INLINE uint32_t NVIC_GetPriority(IRQn_Type IRQn)
787{
788
789 if((int32_t)(IRQn) < 0) {
790 return((uint32_t)(((SCB->SHP[_SHP_IDX(IRQn)] >> _BIT_SHIFT(IRQn) ) & (uint32_t)0xFFUL) >> (8 - __NVIC_PRIO_BITS)));
791 }
792 else {
793 return((uint32_t)(((NVIC->IP[ _IP_IDX(IRQn)] >> _BIT_SHIFT(IRQn) ) & (uint32_t)0xFFUL) >> (8 - __NVIC_PRIO_BITS)));
794 }
795}

802__STATIC_INLINE void NVIC_SystemReset(void)
803{
804 __DSB(); /* Ensure all outstanding memory accesses included
805 buffered write are completed before reset */
806 SCB->AIRCR = ((0x5FAUL << SCB_AIRCR_VECTKEY_Pos) |
807 SCB_AIRCR_SYSRESETREQ_Msk);
808 __DSB(); /* Ensure completion of memory access */
809 while(1) { __NOP(); } /* wait until reset */
810}

840__STATIC_INLINE uint32_t SysTick_Config(uint32_t ticks)
841{
842 if ((ticks - 1UL) > SysTick_LOAD_RELOAD_Msk) {return (1UL);} /* Reload value impossible */
843
844 SysTick->LOAD = (uint32_t)(ticks - 1UL); /* set reload register */
845 NVIC_SetPriority (SysTick_IRQn, (1UL << __NVIC_PRIO_BITS) - 1UL); /* set Priority for Systick Interrupt */
846 SysTick->VAL = 0UL; /* Load the SysTick Counter Value */
847 SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk |
