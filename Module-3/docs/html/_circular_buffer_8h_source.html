43 void push(const T& data) {

46 _tail++;

62 bool pop(T& data) {

64 core_util_critical_section_enter();

67 _tail %= BufferSize;
68 _full = false;

73 }

79 bool empty() {

102 _head = 0;

104 _full = false;
