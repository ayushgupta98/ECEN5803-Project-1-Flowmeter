146 pFunctionPointer_t add_common(T *tptr, void (T::*mptr)(void), IRQn_Type irq, bool front=false) {
147 _mutex.lock();
148 int irq_pos = get_irq_index(irq);
149 bool change = must_replace_vector(irq);
150
151 pFunctionPointer_t pf = front ? _chains[irq_pos]->add_front(tptr, mptr) : _chains[irq_pos]->add(tptr, mptr);
152 if (change)
153 NVIC_SetVector(irq, (uint32_t)&InterruptManager::static_irq_helper);
154 _mutex.unlock();
155 return pf;
156 }
