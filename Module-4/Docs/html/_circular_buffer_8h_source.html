65template<typename T, uint32_t BufferSize, typename CounterType = uint32_t>
66class CircularBuffer {
67public:
68 CircularBuffer() : _head(0), _tail(0), _full(false)
69 {
70 MBED_STATIC_ASSERT(
71 internal::is_unsigned<CounterType>::value,
72 "CounterType must be unsigned"
73 );
74
75 MBED_STATIC_ASSERT(
76 (sizeof(CounterType) >= sizeof(uint32_t)) ||
77 (BufferSize < (((uint64_t) 1) << (sizeof(CounterType) * 8))),
78 "Invalid BufferSize for the CounterType"
79 );
80 }
81
82 ~CircularBuffer()
83 {
84 }

91 void push(const T &data)
92 {
93 core_util_critical_section_enter();
94 if (full()) {
95 _tail++;
96 if (_tail == BufferSize) {
97 _tail = 0;
98 }
99 }
100 _pool[_head++] = data;
101 if (_head == BufferSize) {
102 _head = 0;
103 }
104 if (_head == _tail) {
105 _full = true;
106 }
107 core_util_critical_section_exit();
108 }
109
115 bool pop(T &data)
116 {
117 bool data_popped = false;
118 core_util_critical_section_enter();
119 if (!empty()) {
120 data = _pool[_tail++];
121 if (_tail == BufferSize) {
122 _tail = 0;
123 }
124 _full = false;
125 data_popped = true;
126 }
127 core_util_critical_section_exit();
128 return data_popped;
129 }
130
135 bool empty() const
136 {
137 core_util_critical_section_enter();
138 bool is_empty = (_head == _tail) && !_full;
139 core_util_critical_section_exit();
140 return is_empty;
141 }
142
147 bool full() const
148 {
149 core_util_critical_section_enter();
150 bool full = _full;
151 core_util_critical_section_exit();
152 return full;
153 }
154
158 void reset()
159 {
160 core_util_critical_section_enter();
161 _head = 0;
162 _tail = 0;
163 _full = false;
164 core_util_critical_section_exit();
165 }
166
168 CounterType size() const
169 {
170 core_util_critical_section_enter();
171 CounterType elements;
172 if (!_full) {
173 if (_head < _tail) {
174 elements = BufferSize + _head - _tail;
175 } else {
176 elements = _head - _tail;
177 }
178 } else {
179 elements = BufferSize;
180 }
181 core_util_critical_section_exit();
182 return elements;
183 }
184
190 bool peek(T &data) const
191 {
192 bool data_updated = false;
193 core_util_critical_section_enter();
194 if (!empty()) {
195 data = _pool[_tail];
196 data_updated = true;
197 }
198 core_util_critical_section_exit();
199 return data_updated;
200 }
201
202private:
203 T _pool[BufferSize];
204 CounterType _head;
205 CounterType _tail;
206 bool _full;
207};
