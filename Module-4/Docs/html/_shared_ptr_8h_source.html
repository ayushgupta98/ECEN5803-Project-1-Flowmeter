82 SharedPtr(T *ptr): _ptr(ptr), _counter(NULL)
83 {
84 // Allocate counter on the heap, so it can be shared
85 if (_ptr != NULL) {
86 _counter = new uint32_t;
87 *_counter = 1;
88 }
89 }

106 SharedPtr(const SharedPtr &source): _ptr(source._ptr), _counter(source._counter)
107 {
108 // Increment reference counter
109 if (_ptr != NULL) {
110 core_util_atomic_incr_u32(_counter, 1);
111 }
112 }

120 SharedPtr operator=(const SharedPtr &source)
121 {
122 if (this != &source) {
123 // Clean up by decrementing counter
124 decrement_counter();
125
126 // Assign new values
127 _ptr = source.get();
128 _counter = source.get_counter();
129
130 // Increment new counter
131 if (_ptr != NULL) {
132 core_util_atomic_incr_u32(_counter, 1);
133 }
134 }
135
136 return *this;
137 }

143 void reset(T *ptr)
144 {
145 // Clean up by decrementing counter
146 decrement_counter();
147
148 _ptr = ptr;
149 if (ptr != NULL) {
150 // Allocate counter on the heap, so it can be shared
151 _counter = new uint32_t;
152 *_counter = 1;
153 } else {
154 _counter = NULL;
155 }
156 }

161 void reset()
162 {
163 reset(NULL);
164 }

171 T *get() const
172 {
173 return _ptr;
174 }

180 uint32_t use_count() const
181 {
182 if (_ptr != NULL) {
183 core_util_critical_section_enter();
184 uint32_t current_counter = *_counter;
185 core_util_critical_section_exit();
186 return current_counter;
187 } else {
188 return 0;
189 }
190 }

196 T &operator*() const

199 }

205 T *operator->() const
206 {

208 }
209
214 operator bool() const
215 {
216 return (_ptr != NULL);
217 }
218
219private:
224 uint32_t *get_counter() const
225 {
226 return _counter;
227 }
228
235 void decrement_counter()
236 {
237 if (_ptr != NULL) {
238 uint32_t new_value = core_util_atomic_decr_u32(_counter, 1);
239 if (new_value == 0) {
240 delete _counter;
241 delete _ptr;
242 }
243 }
244 }
245
246private:
247 // Pointer to shared object
248 T *_ptr;
249
250 // Pointer to shared reference counter
251 uint32_t *_counter;
252};
